        \node[stage] (config) at (0,0) {\textbf{1. Configure}\\JSON paths, CSS weights, target browser, limits};

        \draw[optional] (bug.east) -- ++(0.75,0) |- node[near start, above, font=\scriptsize] {No} (pass.west);
        \draw[optional] (pass.north) -- ++(0,2.2) -| (generate.north);
        \draw[flow] (bug.south) -- node[right, font=\scriptsize] {Yes} (sort.north);
        \draw[flow] (sort) -- node[above, font=\scriptsize] {No Match} (minify);
        \draw[optional] (sort.south) -- node[right, align=left, font=\scriptsize] {Match:\\Save in Group} ++(0,-1.05) -| (report.south);
        \draw[flow] (minify) -- (report);
        \draw[optional] (report.north) to[out=105,in=245] node[left, font=\scriptsize] {next test} (generate.south);
    \end{tikzpicture}
    }
